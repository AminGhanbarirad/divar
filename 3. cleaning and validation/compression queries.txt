WHEN title_C like "%خرید%"
OR title_C like "%خراب%"
OR title_C like "%مسافرت%"
OR title_C like "%هتل%"
OR title_C like "%صندوقی%"
OR title_C like "%مینی%"
OR title_C like "%ویترینی%"
OR title_C like "%سوخته%"
--OR title_C like "%تعمیر%"
THEN 1
ELSE 0
END
